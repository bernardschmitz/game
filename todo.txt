/* Several logging routines */

#include <stdio.h>
#include <stdarg.h>

void log_warning(const char *fmt, ...)
{
  va_list ap;
  
  fprintf(stderr, "WARNING: ");
  va_start(ap, fmt);
  vfprintf(stderr, fmt, ap);
  va_end(ap);
  fprintf(stderr, "\n");
}

void log_debug(const char *fmt, ...)
{
#ifdef DEBUG
  va_list ap;
  
  fprintf(stderr, "DEBUG: ");
  va_start(ap, fmt);
  vfprintf(stderr, fmt, ap);
  va_end(ap);
  fprintf(stderr, "\n");
#endif
}
